p5.serial.js

Please connect your Arduino to your computer and upload the following code to it. Then click the "connect" button below to start communicating with your Arduino. 


void setup() {    
    Serial.begin(9600);
    pinMode(4, OUTPUT);
    }
    int count = 0;
    void loop() {
    Serial.write(count);
    delay(100);
    count++;
    if ( count > 100 )count = 0;
    
    if( Serial.available() > 0 ){
        int value = Serial.read();
        if( value == 1){
            digitalWrite(4, HIGH);
        }
        else if( value == 2){
        digitalWrite(4, LOW);
        }
    }
}
